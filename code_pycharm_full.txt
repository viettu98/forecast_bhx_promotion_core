def expand_date_range(df, fromdate_col='fromdate', todate_col='todate'):
    if df.empty:
        return pd.DataFrame(columns=df.columns.to_list() + ['date_key'])

    df[fromdate_col] = pd.to_datetime(df[fromdate_col], errors='coerce')
    df[todate_col] = pd.to_datetime(df[todate_col], errors='coerce')

    df['adjusted_fromdate'] = df[fromdate_col].apply(
        lambda x: x + pd.Timedelta(days=1) if x.hour >= 21 else x
    ).dt.date

    df['adjusted_todate'] = df[todate_col].apply(
        lambda x: x - pd.Timedelta(days=1) if x.hour < 8 else x
    ).dt.date

    df = df.dropna(subset=['adjusted_fromdate', 'adjusted_todate'])

    date_ranges = [pd.date_range(start=row['adjusted_fromdate'], end=row['adjusted_todate'], freq='D')
                   for index, row in df.iterrows()]

    expanded_df = df.loc[df.index.repeat([len(dr) for dr in date_ranges])].reset_index(drop=True)

    expanded_df['date_key'] = [date for dr in date_ranges for date in dr]

    return expanded_df
